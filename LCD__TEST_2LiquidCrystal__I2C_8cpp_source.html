32 // When the display powers up, it is configured as follows:
33 //
34 // 1. Display clear
35 // 2. Function set:
36 // DL = 1; 8-bit interface data
37 // N = 0; 1-line display
38 // F = 0; 5x8 dot character font
39 // 3. Display on/off control:
40 // D = 0; Display off
41 // C = 0; Cursor off
42 // B = 0; Blinking off
43 // 4. Entry mode set:
44 // I/D = 1; Increment by 1
45 // S = 0; No shift
46 //
47 // Note, however, that resetting the Arduino doesn't reset the LCD, so we
48 // can't assume that its in that state when a sketch starts (and the
49 // LiquidCrystal constructor is called).

114 int LiquidCrystal_I2C::i2c_write_byte(int handle, unsigned char val)
115 {
116  if (_rtControl == 0){
117  int i=0;
118  while (write(handle, &val, 1) != 1 && i<10){
119  i++;
120  }
121  if(i>=10)
122  {
123  fprintf(stderr, "i2c_write_byte error: %s\n", strerror(errno));
124  return -1;
125  }
126  }
127  else if(_rtControl == 1) {
128  unsigned char comm = commandQueue.front();
129  if (comm == 0xff){
130  return 0;
131  }
132  int i=0;
133  while (write(handle, &comm, 1) != 1 && i<10){
134  i++;
135  }
136  if(i>=10){
137  fprintf(stderr, "i2c_write_byte error: %s\n", strerror(errno));
138  return -1;
139  }
140  commandQueue.pop();
141  }
142  return(1);
143 }

156 void LiquidCrystal_I2C::begin(uint8_t cols, uint8_t lines, uint8_t dotsize) {
157  if (lines > 1) {
158  _displayfunction |= LCD_2LINE;
159  }
160  _numlines = lines;
161 
162  // for some 1 line displays you can select a 10 pixel high font
163  if ((dotsize != 0) && (lines == 1)) {
164  _displayfunction |= LCD_5x10DOTS;
165  }
166 
167  // SEE PAGE 45/46 FOR INITIALIZATION SPECIFICATION!
168  // according to datasheet, we need at least 40ms after power rises above 2.7V
169  // before sending commands. Arduino can turn on way befer 4.5V so we'll wait 50
170  usleep(50000);
171  // Now we pull both RS and R/W low to begin commands
172  expanderWrite(_backlightval); // reset expanderand turn backlight off (Bit 8 =1)
173  usleep(1000000);
174 
175  //put the LCD into 4 bit mode
176  // this is according to the hitachi HD44780 datasheet
177  // figure 24, pg 46
178 
179  // we start in 8bit mode, try to set 4 bit mode
180  write4bits(0x03 << 4);
181  usleep(4500); // wait min 4.1ms
182 
183  // second try
184  write4bits(0x03 << 4);
185  usleep(4500); // wait min 4.1ms
186 
187  // third go!
188  write4bits(0x03 << 4);
189  usleep(150);
190 
191  // finally, set to 4-bit interface
192  write4bits(0x02 << 4);
193 
194 
195  // set # lines, font size, etc.
196  command(LCD_FUNCTIONSET | _displayfunction);
197 
198  // turn the display on with no cursor or blinking default
199  _displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
200  display();
201 
202  // clear it off
203  clear();
204 
205  // Initialize to default text direction (for roman languages)
206  _displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
207 
208  // set the entry mode
209  command(LCD_ENTRYMODESET | _displaymode);
210 
211  home();
212 
213 }
